    <script>
        const windowEl = document.getElementById('chat-window');
        const inputEl = document.getElementById('user-query');
        const indicator = document.getElementById('typing-indicator');

        // New function to handle Enter key
        function handleEnter(e) {
            if (e.key === 'Enter') {
                sendQuery();
            }
        }

        async function uploadChatFiles(input) {
            if (!input.files.length) return;

            const indicator = document.getElementById('typing-indicator');
            const fileDisplay = document.getElementById('file-list-display');
            const files = input.files;
            const formData = new FormData();
            const currentChips = []; 

            // Create Loading Chips
            for (let i = 0; i < files.length; i++) {
                formData.append('files', files[i]);
                
                const chip = document.createElement('div');
                // Dark theme chip style
                chip.style = "background: #1e293b; padding: 5px 12px; border-radius: 12px; font-size: 12px; color: #93c5fd; border: 1px solid #3b82f6; display: flex; align-items: center; gap: 5px; transition: all 0.3s ease;";
                chip.innerHTML = `<span class="status-icon">⏳</span> ${files[i].name}`;
                fileDisplay.appendChild(chip);
                currentChips.push(chip);
            }

            if (indicator) indicator.style.display = 'block';

            try {
                const res = await fetch("{% url 'chatbot_upload_extra' %}", {
                    method: 'POST',
                    headers: { 'X-CSRFToken': '{{ csrf_token }}' },
                    body: formData
                });

                const data = await res.json();

                if (res.ok) {
                    currentChips.forEach(chip => {
                        chip.style.borderColor = "#22c55e"; 
                        chip.style.background = "rgba(34, 197, 94, 0.1)"; 
                        chip.style.color = "#4ade80"; 
                        chip.querySelector('.status-icon').innerText = "✅";
                    });
                    addMsg(data.message, 'bot');
                } else {
                    throw new Error(data.message);
                }

            } catch (e) {
                currentChips.forEach(chip => {
                    chip.style.borderColor = "#ef4444"; 
                    chip.style.background = "rgba(239, 68, 68, 0.1)"; 
                    chip.style.color = "#f87171"; 
                    chip.querySelector('.status-icon').innerText = "❌";
                });
                addMsg("Upload Error: " + e.message, 'bot');
            } finally {
                if (indicator) indicator.style.display = 'none';
                input.value = '';
            }
        }

        async function sendQuery() {
            const val = inputEl.value.trim();
            if(!val) return;

            addMsg(val, 'user');
            inputEl.value = '';
            indicator.style.display = 'block';

            try {
                const res = await fetch('/chatbot/api/', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json', 'X-CSRFToken': '{{ csrf_token }}' },
                    body: JSON.stringify({ question: val })
                });
                const data = await res.json();
                addMsg(data.answer, 'bot');
            } catch (e) {
                addMsg("Error connecting to server. Please try again.", 'bot');
            } finally {
                indicator.style.display = 'none';
            }
        }

        function addMsg(text, type) {
            const div = document.createElement('div');
            div.className = `msg ${type}`;
            
            if (type === 'bot') {
                // RENDER MARKDOWN HERE
                // marked.parse converts the markdown string to HTML
                div.innerHTML = marked.parse(text);
            } else {
                div.innerText = text;
            }
            
            windowEl.appendChild(div);
            windowEl.scrollTop = windowEl.scrollHeight;
        }
    </script>
